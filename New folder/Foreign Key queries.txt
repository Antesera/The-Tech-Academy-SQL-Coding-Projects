1, CREATE TABLE BORROWER
(
CardNo INT NOT NULL,
Name Varchar(50) NOT NULL,
Address Varchar(50) NOT NULL,
Phone INT NOT NULL,
Primary key (CardNo),
Foreign Key (BORROWER)
);

2, CREATE TABLE BOOKS
(
BookID INT NOT NULL,
Title Varchar(50) NOT NULL,
PublisherName Varchar(50) NOT NULL,
Primary key(BookID),
Foreign Key (BOOK-Authors),
Foreign Key (BOOK-Copies),
Foreign Key (BOOK-LOANS)
);

3, CREATE TABLE PUBLISHER
(
PublisherName Varchar(50) NOT NULL,
Address Varchar(50) NOT NULL,
Phone INT NOT NULL,
Primary key (PublisherName),
Foreign Key (Publisher)
);

4, CREATE TABLE BOOK_AUTHORS
(
BookID INT NOT NULL,
AuthorName Varchar(50) NOT NULL,
);
 
5, CREATE TABLE LIBRARY_BRANCH
(
BranchID INT NOT NULL,
BranchName Varchar(50) NOT NULL,
Address Varchar(50) NOT NULL,
primary key(BranchID),
Foreign Key (LIBRARY_BRANCH),
Foreign Key (BOOK-LOANS)
);

6, CREATE TABLE BOOK_COPIES
(
BookID INT NOT NULL,
BranchID INT NOT NULL,
Number_Of_Copies INT NOT NULL,
);

7, CREATE TABLE BOOK_LOANS
(
BookID INT NOT NULL,
BranchID INT NOT NULL,
CardNo INT NOT NULL,
DateOut INT NOT NULL,
DateDue INT NOT NULL,
);
